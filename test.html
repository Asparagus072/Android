public class MainActivity extends AppCompatActivity {
    // 1. Uporabniški vmesnik [cite: 30]
    EditText editPriimek;
    CalendarView calendarView;
    RadioGroup radioGroup;
    RadioButton rbIzbran;
    ProgressBar progressBar;

    // 2. Seznam in ArrayAdapter [cite: 38]
    ArrayList<String> seznamVnosov;

    // 5. Broadcast Receiver [cite: 41]
    BroadcastReceiver batteryReceiver;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Povezava z XML [cite: 47]
        editPriimek = findViewById(R.id.editPriimek);
        calendarView = findViewById(R.id.calendarView);
        radioGroup = findViewById(R.id.radioGroup);
        progressBar = findViewById(R.id.progressBar);
        listView = findViewById(R.id.listView);
        btnShraniInDodaj = findViewById(R.id.btnShraniInDodaj);

        // Inicializacija seznama [cite: 53]
        seznamVnosov = new ArrayList<>();
        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, seznamVnosov);
        listView.setAdapter(adapter);

        // CalendarView izpis datuma [cite: 57]
        calendarView.setOnDateChangeListener((view, year, month, dayOfMonth) -> {
            Toast.makeText(MainActivity.this, dayOfMonth + "." + (month + 1) + "." + year, Toast.LENGTH_SHORT).show();
        });

        // 3. Shranjevanje in obnova stanja [cite: 62]
        SharedPreferences prefs = getSharedPreferences("MojeNastavitveB", MODE_PRIVATE);
        String shranjenPriimek = prefs.getString("priimek", "");
        int shranjenRadioId = prefs.getInt("radioIzbira", -1);
        editPriimek.setText(shranjenPriimek);
        if (shranjenRadioId != -1) { radioGroup.check(shranjenRadioId); }

        // Klik na gumb [cite: 71]
        btnShraniInDodaj.setOnClickListener(v -> {
            String priimek = editPriimek.getText().toString();
            int selectedId = radioGroup.getCheckedRadioButtonId();
            String radioTekst = "";
            if (selectedId != -1) {
                rbIzbran = findViewById(selectedId);
                radioTekst = rbIzbran.getText().toString();
            }
            seznamVnosov.add(priimek + " - " + radioTekst);
            adapter.notifyDataSetChanged();

            SharedPreferences.Editor editor = prefs.edit();
            editor.putString("priimek", priimek);
            editor.putInt("radioIzbira", selectedId);
            editor.apply();
        });

        // 5. Broadcast Receiver za baterijo [cite: 89]
        batteryReceiver = new BroadcastReceiver() {
            @Override
            public void onReceive(Context context, Intent intent) {
                Toast.makeText(context, "Nizka baterija!", Toast.LENGTH_LONG).show();
            }
        };
    }

    // 4. Gnezdeni opcijski meni [cite: 97]
    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        getMenuInflater().inflate(R.menu.options_menu, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(MenuItem item) {
        int id = item.getItemId();
        if (id == R.id.menu_verzija) {
            Toast.makeText(this, "Verzija 1.0", Toast.LENGTH_SHORT).show();
            return true;
        } else if (id == R.id.menu_barva) {
            editPriimek.setTextColor(Color.RED);
            return true;
        } else if (id == R.id.menu_velikost) {
            editPriimek.setTextSize(24f);
            return true;
        }
        return super.onOptionsItemSelected(item);
    }

    @Override
    protected void onStart() {
        super.onStart();
        registerReceiver(batteryReceiver, new IntentFilter(Intent.ACTION_BATTERY_LOW));
    }

    @Override
    protected void onStop() {
        super.onStop();
        unregisterReceiver(batteryReceiver);
    }
}
